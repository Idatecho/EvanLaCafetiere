from settings import *


class Camera:
    def __init__(self, position, yaw, pitch):
        # Initialise la position de la cam√©ra comme un vecteur 3D
        self.position = glm.vec3(position)

        # Yaw : rotation horizontale, convertie en radians
        self.yaw = glm.radians(yaw)

        # Pitch : rotation verticale, convertie en radians
        self.pitch = glm.radians(pitch)

        # Vecteur haut (vers le haut de la sc√®ne)
        self.up = glm.vec3(0, 1, 0)

        # Vecteur droite (horizontal vers la droite)
        self.right = glm.vec3(1, 0, 0)

        # Vecteur avant (direction de vue initiale)
        self.forward = glm.vec3(0, 0, -1)

        # Matrice de projection en perspective
        self.m_proj = glm.perspective(V_FOV, ASPECT_RATIO, NEAR, FAR)

        # Matrice de vue (sera mise √† jour plus tard)
        self.m_view = glm.mat4()


    def update(self):
        # Met √† jour les vecteurs directionnels selon yaw/pitch
        self.update_vectors()

        # Met √† jour la matrice de vue selon la nouvelle orientation
        self.update_view_matrix()


    def update_view_matrix(self):
        # Calcule la matrice de vue avec glm.lookAt : position, cible, et vecteur "up"
        self.m_view = glm.lookAt(self.position, self.position + self.forward, self.up)


    def update_vectors(self):
        # Calcule le vecteur forward √† partir des angles yaw et pitch
        self.forward.x = glm.cos(self.yaw) * glm.cos(self.pitch)
        self.forward.y = glm.sin(self.pitch)
        self.forward.z = glm.sin(self.yaw) * glm.cos(self.pitch)

        # Normalise le vecteur forward
        self.forward = glm.normalize(self.forward)

        # Calcule le vecteur right perpendiculaire √† forward et au haut global
        self.right = glm.normalize(glm.cross(self.forward, glm.vec3(0, 1, 0)))

        # Calcule le vecteur up perpendiculaire √† right et forward
        self.up = glm.normalize(glm.cross(self.right, self.forward))


    def rotate_pitch(self, delta_y):
        # Modifie l'angle vertical (pitch) selon le mouvement de la souris
        self.pitch -= delta_y

        # Limite l'angle pitch pour √©viter les retournements de cam√©ra
        self.pitch = glm.clamp(self.pitch, -PITCH_MAX, PITCH_MAX)


    def rotate_yaw(self, delta_x):
        # Modifie l'angle horizontal (yaw) selon le mouvement de la souris
        self.yaw += delta_x


    def move_left(self, velocity):
        # D√©place la cam√©ra vers la gauche (strafe)
        self.position -= self.right * velocity


    def move_right(self, velocity):
        # D√©place la cam√©ra vers la droite (strafe)
        self.position += self.right * velocity


    def move_up(self, velocity):
        # D√©place la cam√©ra vers le haut
        self.position += self.up * velocity


    def move_down(self, velocity):
        # D√©place la cam√©ra vers le bas
        self.position -= self.up * velocity


    def move_forward(self, velocity):
        # Avance dans la direction actuelle de la cam√©ra
        self.position += self.forward * velocity


    def move_back(self, velocity):
        # Recule dans la direction oppos√©e √† la vue
        self.position -= self.forward * velocity
